;; PRODUCTIONS - let's try to keep them organized
;; GENERAL

(chunk-type position pos-x pos-y)
(chunk-type environment jameson-position-x jameson-position-y oldest-position-x oldest-position-y middle-position-x middle-position-y newest-position-x newest-position-y state)
(chunk-type projectile id environment)
(chunk-type estimation dx dy y action)

(add-dm
  (goal isa environment)(attended)(attending)(done)(pos-encoded)
  (pos1-encoded)(pos2-encoded)(pos3-encoded)(estimate)(retrieving))



;; VISUAL-MODULE PRODUCTION

(P attend-projectile
   =goal>
      ISA         environment
	  state				nil
   =visual-location>
   		screen-x				=pos-x
   		screen-y				=pos-y
   ?visual>
      state       free
   ?imaginal>
   		state				free
   
==>
   +visual>
      cmd         move-attention
      screen-pos  =visual-location
   =goal>
      state       nil
   +imaginal>
   		pos-x				=pos-x
   		pos-y				=pos-y
)


(P encode-trajectory-one
		=goal>
			isa						environment
			state	nil
		?imaginal>
			buffer				full
		=imaginal>
			pos-x= pos-x
			pos-y= pos-y
		?manual>
			state					free
==>
		=goal>
			oldest-position-x			=pos-x
			oldest-position-y			=pos-y
			state	pos1-encoded
   	+manual>
  		cmd       	  press-key
  		key       	  "s"
  	+imaginal>
  	
)
(P encode-trajectory-two
		=goal>
			isa						environment
			state	pos1-encoded
		?imaginal>
			buffer				full
		=imaginal>
			pos-x= pos-x
			pos-y= pos-y
		?manual>
			state					free
==>
		=goal>
			middle-position-x			=pos-x
			middle-position-y			=pos-y
			state	pos2-encoded
		+imaginal>
   	+manual>
  		cmd       	  press-key
  		key       	  "s"
)
(P encode-trajectory-three
		=goal>
			isa						environment
			state	pos2-encoded
		?imaginal>
			buffer 				full
		=imaginal>
			pos-x= pos-x
			pos-y= pos-y
		?manual>
			state 				free
==>
		=goal>
			newest-position-x		  =pos-x
			newest-position-y		  =pos-y
			state  estimate
   	+manual>
  		cmd    	     press-key
  		key    	     "s"
  	+imaginal>
  	
)
(P estimate-course
		=goal>
			isa						environment
			state	estimate		
			jameson-position-y			=j-y
			middle-position-x			=p2-x
			middle-position-y			=p2-y
			newest-position-x			=p3-x
			newest-position-y			=p3-y
			
==>
		!output!				"Estimating trajectory"
		=goal>
			state	retrieving
		+retrieval>
			isa				estimation
			dx= (- p3-x p2-x)
			dy= (- p3-y p2-y) 
			y=j-y
)

(P remmeber-situation
    =goal>
     isa    environment
     state     retrieving
    =retireval>
     isa    estimation
     action =act
    ?manual>
     state free
   ==>
    =goal>
      state nil
    +manual>
     cmd press-key
     key =act
     
    @retrieval>)

(P cant-remember-up
    =goal>
      isa environment
      state retrieving
	  jameson-position-y	=j-y
    ?retrieval>
      buffer failure
	=retireval>
      = y (+ j-y dy)
      <= dy 0
    ?manual>
      state free
   ==>
     =goal>
       state nil
     +manual>
       cmd press-key
       key "u")

(P cant-remember-down

	=retireval>

     +manual>

       key "d")

(P cant-remember-stay
    =goal>
      isa environment
      state retrieving
	  jameson-position-y	=j-y
    ?retrieval>
      buffer failure
	=retireval>
      - y  (+ j-y dy)
    ?manual>
      state free
   ==>
     =goal>
       state nil
     +manual>
       cmd press-key
       key "s")
